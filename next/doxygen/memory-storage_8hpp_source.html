26struct mem_key_t{

28 size_t local_id:16=0; //Unique local id identifying the position of this resource in location. If left "" it represents the full file.
29 resource_t resource:4=resource_t::NONE;
30 size_t to_promote:1=false; //Consider for promotion to higher tier cache
